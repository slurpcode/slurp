def check(path)
  Dir.glob("#{path}/**/*.{xml,xsd,xsl}").each do |filename|
    doc = File.open(filename) {|xml| Nokogiri.XML(xml)}
    puts filename, doc.errors unless doc.errors.empty?
  end
end

def get_energy_points(user)
  uri = URI.parse("https://www.khanacademy.org/profile/#{user}/")
  doc = Nokogiri.HTML(uri.open)
  res =
    doc.xpath('//script[contains(.,"prefersReducedMotion")]').to_s.scan(
      /points":(\d+),"prefersReducedMotion/
    )
  res.size.positive? ? res : 'Not found'
end

    strip_white_space() click to toggle source

def strip_white_space
  Dir.glob('output/**/*.*').each do |filename|
    file = File.open(filename, 'r+')
    contents = file.read
    strip_contents = contents.strip
    file.rewind
    file.truncate(0)
    file.write(strip_contents)
    file.close
  end
end
